[docs]def make_Poly(save=_save, path=_here):
    notes = get_notes()

    lPoly = []
    for k in notes.keys():
        if k[-2] in ['1','2']:
            continue
        Poly = np.array([np.hypot(notes[k][:,0],notes[k][:,2]), notes[k][:,1]])
        lPoly.append(Poly)
        if save:
            cls = 'CoilPF'
            name = '%sV0'%k
            savename = 'TFG_%s_ExpWEST_%s'%(cls,name)
            pathfilext = os.path.join(path, savename+'.txt')

            npno = np.r_[Poly.shape[1], 0]
            savepoly = np.vstack((npno,Poly.T))
            np.savetxt(pathfilext, savepoly, comments='#',
                       header='Name = %s\nExp = %s\nCLs = %s'%(name,'WEST',cls))
    return lPoly, notes
