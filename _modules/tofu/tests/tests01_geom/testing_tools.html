    ves_norm = ves_poly[:, 1:] - ves_poly[:, :-1]
    ves_norm = np.array([-ves_norm[1, :], ves_norm[0, :]])
    ves_norm = ves_norm / np.sqrt(np.sum(ves_norm ** 2, axis=0))[np.newaxis, :]
