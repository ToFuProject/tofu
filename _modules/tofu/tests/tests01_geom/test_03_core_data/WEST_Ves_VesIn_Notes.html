    [docs]def make_Poly(save=_save, path=_here):

    notes = get_notes()

    C = notes['C']
    nP = 100
    theta = np.linspace(0.,2*np.pi, nP, endpoint=False)
    P = np.array([C[0]+notes['r_in']*np.cos(theta),
                  C[1]+notes['r_in']*np.sin(theta)])

    if save:
        cstr = '%s_%s_%s'%(_Exp,_Cls,_name)
        pathfilext = os.path.join(path, cstr+'_V0.txt')
        np.savetxt(pathfilext, P)
    return P, notes
